class ConvNet(nn.Module):
    def __init__(self):
        super(ConvNet,self).__init__()
        self.conv1 = nn.Conv2d(3,64,3)
        self.pool1 = nn.MaxPool2d(3,stride=1)
        self.conv2 = nn.Conv2d(64,128,4)
        self.pool2 = nn.MaxPool2d(3,stride=1)
        self.conv3 = nn.Conv2d(128,32,4)
        self.pool3 = nn.MaxPool2d(2,stride=2)
        self.conv4 = nn.Conv2d(32,16,2)
        self.droput = nn.Dropout(0.15)
        self.fc1 = nn.Linear(1024,128)
        self.fc2 = nn.Linear(128,32)
        self.fc3 = nn.Linear(32,10)
        self.criterion = nn.CrossEntropyLoss()
        self.optimizer = None

    def forward(self,x):
        x = self.pool1(F.elu(self.conv1(x)))
        x = self.pool2(F.elu(self.conv2(x)))
        x = self.pool2(F.elu(self.conv3(x)))
        x = self.droput(x)
        x = self.pool3(F.elu(self.conv4(x)))
        x = self.droput(x)
        x = torch.flatten(x, 1) # flatten all dimensions except batch
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        x = self.fc3(x)
        return x

    def save(self, model_name):
        if not os.path.exists(".models/"):
            os.makedirs(".models/")
        PATH = f'.models/{model_name}.pth'
        torch.save(self.state_dict(), PATH)

    def load(self, model_name):
        PATH = f'.models/{model_name}.pth'
        self.load_state_dict(torch.load(PATH))

    def set_optimizer(self):
        self.optimizer = torch.optim.SGD(self.parameters(), lr=LR)


    def train_model(self,starting_epoch=0,total_epochs=EPOCHS):
        for epoch in range(starting_epoch,total_epochs):
            accuracy = 0
            total_loss = 0
            for (images,labels) in tqdm(train_loader):
                outputs = self(images)
                self.optimizer.zero_grad()
                loss = self.criterion(outputs,labels)
                loss.backward()
                self.optimizer.step()

                with torch.no_grad():
                    total_loss += loss/NUM_OF_BATCHES
                    _, predicted = torch.max(outputs,dim=1)
                    accuracy += (torch.eq(predicted,labels)).sum().item()/train_size

            print(f'epoch {epoch + 1} had {100*accuracy:.2f}% accuracy and {total_loss:.3f} loss')
